//TO SORT A SET OF 25 NOs. BY USING INSERTION SORT
#include<stdio.h> /*TO INCLUDE HEADER FILES*/
#include<conio.h>
#define MAX 5 /*MAX IS DEFINED AS 5*/
void main() /*BY WRITING VOID,MAIN DOES DOES NOT RETURN ANY VALUE*/
{
int nu[MAX],i,j,c; /*DECLARATION PART*/
clrscr();
printf("ENTER %d NOs.: ",MAX);
for(i=0;i<MAX;i++)
scanf("%d",&nu[i]); /*STORES VALUE IN ARRAY*/
for(i=1;i<MAX;i++)
{
if(nu[i]<nu[0])
{
for(j=i;j>0;j--)
{
c=nu[j];
nu[j]=nu[j-1];
nu[j-1]=c;
} /*END OF FOR*/
}
} /*END OF FOR*/
printf("AFTER SORTING:\n");
for(i=0;i<MAX;i++)
printf("\t\t\t%d\n",nu[i]);
getch(); /*PRINT RESULT ON SCREEN*/
}
