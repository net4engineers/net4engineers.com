//TO SORT A SET OF 25 NOs. BY USING SELECTION SORT
#include<stdio.h>
#include<conio.h>
#define MAX 5
void main()
{
int nu[MAX],i,j,a;
clrscr();
printf("ENTER %d NOs.: ",MAX);
for(i=0;i<MAX;i++)
scanf("%d",&nu[i]);
for(i=0;i<MAX;i++)
{
for(j=0;j<MAX;j++)
{
if(j>i)
{
if(nu[j]<nu[i])
{
a=nu[j];
nu[j]=nu[i];
nu[i]=a;
}
}
}
}
printf("AFTER SELECTION SORTING:\n");
for(i=0;i<MAX;i++)
printf("\t\t\t%d\n",nu[i]);
getch();
}
