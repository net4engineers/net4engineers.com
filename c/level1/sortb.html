//TO SORT A SET OF 25 NOs. BY USING BUBBLE SORT
#include<stdio.h>
#include<conio.h>
#define MAX 5
void main()
{
int nu[MAX],i,j,c;
clrscr();
printf("ENTER %d NOs.: ",MAX);
for(i=0;i<MAX;i++)
scanf("%d",&nu[i]);
for(i=0;i<MAX-1;i++)
{
for(j=i+1;j<MAX;j++)
{
if(nu[i]>nu[j])
{
c=nu[i];
nu[i]=nu[j];
nu[j]=c;
}
}
}
printf("AFTER SORTING:\n");
for(i=0;i<MAX;i++)
printf("\t\t\t%d\n",nu[i]);
getch();
}
