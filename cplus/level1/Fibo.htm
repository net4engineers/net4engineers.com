/* fibo.cpp -- It prints out the first N Fibonacci
* numbers.
*/

#include <iostream>
#include <iomanip>

int main(void) {
int n; // The number of fibonacci numbers we will print 
int current; // The value of the (i)th fibonacci number 
int next; // The value of the (i+1)th fibonacci number 
int twoaway; // The value of the (i+2)th fibonacci number 

cout << "How many Fibonacci numbers do you want to compute? ";
cin >> n;
if (n<=0)
cout << "The number should be positive." << endl;
else {
cout << "\n\n\t I \t Fibonacci(I) \n\t=====================" << endl;
next = current = 1;
for (int i=1; i<=n; i++) {
cout << "\t" << setw(3) << i << "\t\t" 
<< setw(4) << current << endl;
twoaway = current+next;
current = next;
next = twoaway;
}
}
}
