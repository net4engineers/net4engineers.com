Task :- Read an integer array, print it, then sort it and print it. Using  the bubble sort method.

#include <iostream.h>
#include <iomanip.h>

void printIntArray(int a[], int n)
/* n is the number of elements in the array a.
* These values are printed out, five per line. */
{
for (int i = 0; i < n; ){
cout << setw(10) << a[i++];
if (i%5==0)
cout << endl;
}
cout << endl;
}

int getIntArray(int a[], int nmax, int sentinel)
/* It reads up to nmax integers and stores then in a; sentinel 
* terminates input. */
{
int n = 0;
int temp;

while (1) {
cout << "Enter integer [" << sentinel << " to terminate] : ";
cin >> temp;
if (temp == sentinel) break;
if (n == nmax)
cout << "array is full\n";
else 
a[n++] = temp;
}
return n;
}

void bubbleSort(int a[], int n)
/* It sorts in non-decreasing order the first N positions of A. It uses
* the bubble sort method.
*/
{
int limit = n-1;
int lastChange;

while (limit) {
lastChange = 0;
for (int temp, lcv = 0; lcv < limit; lcv++)
/* Notice that the values in positions LIMIT+1 .. N are in
* their final position, i.e. they are sorted right */
if (a[lcv] > a[lcv+1]) {
temp = a[lcv];
a[lcv] = a[lcv+1];
a[lcv+1] = temp;
lastChange = lcv;
}
limit = lastChange;
}
}

enum {NMAX = 10};

int main(void) {
int x[NMAX];
int hmny;

hmny = getIntArray(x, NMAX, 0);
if (hmny == 0)
cout << "This is the empty array!" << endl;
else{
cout << "The array was: " << endl;
printIntArray(x,hmny);
bubbleSort(x,hmny);
cout << "The sorted array is: " << endl;
printIntArray(x,hmny);
}
}
