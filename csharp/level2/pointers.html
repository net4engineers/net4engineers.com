using System;

class Test
{
unsafe static void WriteLocations() 
{
int *p;
int i;
i = 10;
p = &i;


string addr;
addr = int.Format((int) p, "X");
//To Format the Pointer to a String

Console.WriteLine(addr);
Console.WriteLine(*p);

*p = 333;
Console.WriteLine(i);
//Change the Value using Pointer

i = *(&i) + 10;

}

static void Main() 
{
WriteLocations();
}
