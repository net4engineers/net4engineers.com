A code sample of how to make an API call using C#. This program enables you to find out which drive letter is the removable, fixed, remote or a CD-ROM Drive 

sing System;
using System.Runtime.InteropServices;
class Test
{
    public enum DriveType
    {
        DRIVE_UNKNOWN = 0,
        DRIVE_NO_ROOT_DIR = 1,
        DRIVE_REMOVABLE = 2,
        DRIVE_FIXED = 3,
        DRIVE_REMOTE = 4,
        DRIVE_CDROM = 5,
        DRIVE_RAMDISK = 6
    }
    
    static public void Main(string[] args)
    {
        DriveType driveType = Test.GetDriveType(args[0]);
        Console.WriteLine("Drive {0} is type {1}", args[0], driveType);
    }

    [DllImport("kernel32.dll")]
    public static extern DriveType GetDriveType(string rootPathName);
}
