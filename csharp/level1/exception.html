// Program to illustrate the concept of exception handling

using System;

class CThrowException
{
public static void Main(string[] arg)
{
// see if any command line arguments have been
// specified or not...
try
{
if (arg.Length==0)
throw new System.Exception();
else
Console.WriteLine("Command Line Argument: {0}",arg[0]);
}
catch(System.Exception e)
{
Console.WriteLine("No command line argument specified. Please do so.");
}
finally
{
Console.WriteLine("Execution Over!");
}

}
}
