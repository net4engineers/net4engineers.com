// Program to illustrate the concept of hiding methods

using System;

class CBase
{
public void MyMethod()
{
Console.WriteLine("This is the Base class");
}
}

class CChild:CBase
{
new public void MyMethod()
{
Console.WriteLine("This is the Child class");
}
}

class CRunApp
{
public static void Main()
{
CChild child = new CChild();
child.MyMethod(); // call method from child object
CBase bbase = new CChild();
bbase.MyMethod(); // call method from base object
((CBase)child).MyMethod(); // what do u think this will do ?
}
}
