// Program to illustrate the concept of arrays in C#

using System;

class CArray
{
public static void Main()
{
int [] sd; // single dimensional array
int [,] td; // two dimensional array
int [][] jg; // jagged array

// single dimensional array...
sd = new int[5];
Console.WriteLine("Array SD has the dimension: {0} and length: {1}",sd.Rank,sd.Length);
sd[0]=1; // assign some value...

// rectangular two dimensional array...
td = new int[5,4];
Console.WriteLine("Array TD has the dimension: {0} and length: {1}",td.Rank,td.Length);
td[0,0]=1; // assign yet another value...

// jagged multidimensional array...
jg = new int[2][];
jg[0] = new int[3];
jg[1] = new int[4];
Console.WriteLine("Array JG has the dimension: {0} and length: {1}",jg.Rank,jg.Length);
jg[0][0]=1; // assign yet another value...

}
}
